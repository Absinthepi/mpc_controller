cmake_minimum_required(VERSION 3.34)
project(mpc_controller)

# 查找 ROS2 依赖
find_package(ament_cmake REQUIRED)
find_package(rclcpp REQUIRED)
find_package(ackermann_msgs REQUIRED)
find_package(nav_msgs REQUIRED)
find_package(tf2 REQUIRED)
find_package(geometry_msgs REQUIRED)
find_package(eigen3_cmake_module REQUIRED)
find_package(Eigen3 REQUIRED)

# 强制使用 /usr/local 下的 Ipopt (3.14.20)
set(IPOPT_ROOT /usr/local)
find_library(IPOPT_LIB ipopt
  PATHS ${IPOPT_ROOT}/lib
  NO_DEFAULT_PATH
  REQUIRED)
find_path(IPOPT_INCLUDE_DIR IpIpoptApplication.hpp
  PATHS ${IPOPT_ROOT}/include/coin-or
  NO_DEFAULT_PATH
  REQUIRED)

# 强制使用 /usr 下的 CppAD（系统一般只有一套，不会冲突）
find_package(PkgConfig REQUIRED)
pkg_check_modules(CPPAD REQUIRED cppad)

# 包含头文件目录
include_directories(
  include
  ${EIGEN3_INCLUDE_DIR}
  ${IPOPT_INCLUDE_DIR}
  ${CPPAD_INCLUDE_DIRS}
)

# 添加可执行文件
add_executable(mpc_controller_node
  src/mpc_controller_node.cpp
  src/MPC.cpp
)

# 链接库
target_link_libraries(mpc_controller_node
  ${IPOPT_LIB}
  ${CPPAD_LIBRARIES}
)

ament_target_dependencies(mpc_controller_node
  rclcpp ackermann_msgs nav_msgs tf2 geometry_msgs
)

# 将 /usr/local/lib 写入可执行文件的 RPATH，确保运行时能找到正确的 .so
target_link_options(mpc_controller_node PRIVATE
  -Wl,-rpath,/usr/local/lib
)

install(TARGETS mpc_controller_node
  DESTINATION lib/${PROJECT_NAME})

ament_package()
